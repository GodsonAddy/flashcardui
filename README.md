# flashcardui

> UI library for flash card application

## Install

```bash
npm i flashcardui
```

## Description

UI library for flash card application

## Usage

```tsx
import React, { Component } from 'react';

import { Button } from 'flashcardui';

class Example extends Component {
  render() {
    return (
      <Button
        color="#ff7043"
        label="flashUI"
        hover="#bf360c"
        buttonType="contained"
      />
    );
  }
}
```
